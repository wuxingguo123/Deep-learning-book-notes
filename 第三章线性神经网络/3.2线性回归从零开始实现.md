```python
#Creat_data为创建相关数据集
def Creat_data(num_input,w,b):
    X = torch.normal(0 , 1 ,(num_input , len(w)))
    Y = torch.matmul(X , w) + b
    Y+=torch.normal(0 , 0.01 , Y.shape)#这里加的是白噪音，一个略微的误差
    return  X,Y.reshape(-1 , 1)
```
